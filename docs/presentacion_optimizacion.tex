% SLIDE 3: Mejoras por Eje Estratégico
% =====================================================================
\begin{frame}{Mejoras por Eje Estrat\'egico}

\begin{columns}[T]
\begin{column}{0.48\textwidth}

\textcolor{blue}{\faIcon{bolt}\ \textbf{Velocidad}}
\begin{itemize}\small
\item Pipeline paralelo (ThreadPoolExecutor, 4 workers)
\item PDF se abre 1 vez (antes 3 veces)
\item Vendor matching 10$\times$ m\'as r\'apido (rapidfuzz)
\item M\'etricas via SQL GROUP BY (elimina N+1)
\item Upload async (no bloquea event loop)
\end{itemize}

\vspace{8pt}
\textcolor{green}{\faIcon{crosshairs}\ \textbf{Precisi\'on}}
\begin{itemize}\small
\item RFC centralizado con check digit + correcci\'on OCR
\item Clasificador muestrea 5 p\'ags con votaci\'on por mayor\'ia
\item ocrmypdf: deskew + clean autom\'atico
\item Extracci\'on de tablas en PDFs digitales
\item Deduplicaci\'on por SHA-256
\end{itemize}

\end{column}
\begin{column}{0.48\textwidth}

\textcolor{amber}{\faIcon{expand-arrows-alt}\ \textbf{Escalabilidad}}
\begin{itemize}\small
\item OCR con l\'imite de 50 p\'ags + gc.collect
\item Liberaci\'on de memoria p\'agina por p\'agina
\item Paginaci\'on, filtros y orden en API
\item Health checks: BD, Tesseract, disco
\item Alembic para migraciones de schema
\end{itemize}

\vspace{8pt}
\textcolor{red}{\faIcon{search}\ \textbf{Auditabilidad}}
\begin{itemize}\small
\item \texttt{DocumentProcessingLog} --- historial completo
\item \texttt{error\_category} indexado y filtrable
\item \texttt{field\_confidence} persistida como JSON
\item \texttt{processing\_route} traza el camino exacto
\item Logging estructurado en todos los m\'odulos
\end{itemize}

\end{column}
\end{columns}

\end{frame}
